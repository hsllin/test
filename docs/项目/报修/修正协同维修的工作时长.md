问题引起原因：
在完成维修的时候，根据协同维修工的数量copy相同数量的工时时长，这个没问题，但是在更新工时时长的时候把woker_id更新为主维修工了，导致出现了

```sql
DELETE from QYH_BU_REPAIR_WORK_TIME where IS_COLLABOTATE_ORDER='1' ;

		 INSERT INTO QYH_BU_REPAIR_WORK_TIME(ID, REPAIR_ORDER_ID, WORKER_ID, ORDER_FINISH_TIME, WORKER_RECEIVE_TIME, WORKER_FINISH_TIME, ORDER_CREATE_TIME, COUNT_WORKER_TIME, COUNT_FINISH_TIME, IS_COLLABOTATE_ORDER)
		 select SUBSTR(TO_CHAR(DBMS_RANDOM.VALUE(11, 99)), 4, 32) as ID, t.REPAIR_ORDER_ID, c.WORKER_ID, ORDER_FINISH_TIME, WORKER_RECEIVE_TIME, WORKER_FINISH_TIME, ORDER_CREATE_TIME, COUNT_WORKER_TIME, COUNT_FINISH_TIME, IS_COLLABOTATE_ORDER from QYH_BU_REPAIR_COLLABORATE c   
		 join QYH_BU_REPAIR_WORK_TIME t on t.REPAIR_ORDER_ID=c.ORDER_ID and t.WORKER_ID!=c.WORKER_ID;
```
